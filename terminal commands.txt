cd [название папки] --перейти в указанную папку
cd .. --вернуться на одну директорию назад
cd ../../../ --вернуться на указанное количество директорий
cd ./название папки --указатель директории
cd "название папки с пробелами" --перейти в папку с названием с пробелами
cd название папки\с пробелами --перейти в папку с названием с пробелами используя экранировние пробелов (обратный слэш)

mkdir [название папки] --создание новой директории (чтобы создать несколько папок ввести названия папок подряд)
mkdir [назавние папки/название папки] --создать в существующей папке новую папку
mkdir -p [назавние папки/название папки] --создать все папки, которые необходимо
mkdir -p {название папки/назавние папки, название папки/назавние папки} --создать несколько директорий со вложенностью
rmdir [название директории] --удалить указанную директорию
rm -r [название директории] --удалить указанную директорию, содержащую файлы

touch [название файла.расширение] --создание нового файла с указанным расширением
rm [название файла.расширение] --удалить указанный файл
 
ls --список всех файлов в выбранной директории
    ls -l --указать файлы в виде списка
    ls -l -a --указать все файлы и ссылки, который есть в директории
    ls -la --указать все файлы и ссылки, который есть в директории
    ls -la > название файла.расширение --поместить вывод команды в указанный файл
pwd --полный адрес текущей директории
clear --очистка консоли

mv [название файла] [название папки] --переместить указанный файл в указанную папку
mv [название файла] [название файла] --переименовать указанный первым файл
mv *.html [название файла] --переместить все файлы с указанным расширением
cp [название файла] [название папки] --скопировать указанный файл в указанную папку
cp - r [название папки] [название папки] --скопировать указанную папку в указанную папку

find --поиск файлов по различным параметрам
